def clear_link_cache(cache: nil, cache_options: {})
  key_prefix = resolve_opengraph_key_prefix(cache_options)
  links = resolve_links
  links.each do |url|
    delete_opengraph_cache_entry(url, cache, cache_options, key_prefix)
  end
end
