def link_objects(with_opengraph: false, cache: nil, cache_options: {})
  if with_opengraph
    links.map { |url| { url: url, opengraph: extract_opengraph(url, cache: cache, cache_options: cache_options) } }
  else
    links.map { |url| { url: url } }
  end
end
